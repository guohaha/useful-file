//获取天数
	private function get_day($date1,$date2){
	    $zero1=$date1; //当前时间  ,注意H 是24小时 h是12小时
	    $zero2=$date2;  //未来时间，不能写2014-1-21 24:00:00  这样不对
	    $guonian=ceil(($zero2-$zero1)/86400); //60s*60min*24h，向上取整
	    return  $guonian;
	}
